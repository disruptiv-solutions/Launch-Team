rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {
    
    // Session-related files (attachments and generated audio)
    match /sessions/{sessionId}/{allPaths=**} {
      // Allow read access to all session files
      allow read: if true;
      
      // Allow write access to all session files
      // NOTE: For production, you should add authentication checks:
      // allow write: if request.auth != null;
      allow write: if true;
    }

    // Project-related files (project documents and project chat attachments)
    match /projects/{projectId}/{allPaths=**} {
      allow read: if true;
      allow write: if true;
    }

    // Default rule for everything else
    match /{allPaths=**} {
      allow read, write: if false;
    }
  }
}
